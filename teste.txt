
SWITCH CASE
let fruta = "a"

switch(fruta){
    case "laranja":
        console.log('Suco de laranja')
        break
    case "banana":
        console.log('Vitamina de banana')
        break
    case "maça":
        console.log('Suco de maça')
    default:
        console.log('Você quer água')
}
